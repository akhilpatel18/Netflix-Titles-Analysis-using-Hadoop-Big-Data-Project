 #!/usr/bin/env python
       import sys
       current = None
       count = 0

        for line in sys.stdin:
         key, value = line.strip().split('\t')
         value = int(value)
                             
          if current == key:
            count += value
          else:
              If current:
                   print("%s\t%d" % (current, count))
                   current = key
                   count = value

                   if current:
                   print("%s\t%d" % (current, count))
